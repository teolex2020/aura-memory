// @aura/contract/src/EpistemicRuntime.ts — 更新后的 Interface

export interface EpistemicRuntimeImpl {
  // ── Belief 层 (6 methods) ──

  /** 按状态过滤 beliefs (Resolved/Unresolved/Singleton/Empty) */
  getBeliefs: (
    stateFilter?: string
  ) => Effect.Effect<ReadonlyArray<Belief>, never, BeliefEngine>

  /** 查 record 对应的 belief */
  getBeliefForRecord: (
    recordId: string
  ) => Effect.Effect<Belief | null, never, BeliefEngine>

  /** 高波动 beliefs (volatility >= threshold, 按 volatility desc 排序) */
  getHighVolatilityBeliefs: (
    minVolatility?: number,  // default: 0.20, clamp [0, 1]
    limit?: number           // default: 20, max: 100
  ) => Effect.Effect<ReadonlyArray<Belief>, never, BeliefEngine>

  /** 低稳定性 beliefs (stability <= threshold, 按 stability asc 排序) */
  getLowStabilityBeliefs: (
    maxStability?: number,   // default: 1.0, max: 1.0
    limit?: number           // default: 20, max: 100
  ) => Effect.Effect<ReadonlyArray<Belief>, never, BeliefEngine>

  /** 信念健康度聚合摘要 (包含 volatility bands, contradiction count) */
  getBeliefInstabilitySummary: () =>
    Effect.Effect<BeliefInstabilitySummary, never, BeliefEngine>

  /** 图连通分量 — 找出共享 record/tag 的冲突信念簇 */
  getContradictionClusters: (
    namespace?: string,
    limit?: number           // default: 20, max: 100
  ) => Effect.Effect<ReadonlyArray<ContradictionCluster>, never, BeliefEngine>

  // ── Concept 层 (4 methods) ──

  /** 按状态过滤 concepts (Stable/Candidate/Rejected) */
  getConcepts: (
    stateFilter?: string
  ) => Effect.Effect<ReadonlyArray<ConceptCandidate>, never, ConceptEngine>

  /** 对外暴露的重要 concept (带 telemetry 计数) */
  getSurfacedConcepts: (
    limit?: number
  ) => Effect.Effect<ReadonlyArray<SurfacedConcept>, never, ConceptEngine>

  /** 按 namespace 过滤 surfaced concepts */
  getSurfacedConceptsForNamespace: (
    namespace: string,
